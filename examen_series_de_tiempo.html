import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

import os
import warnings
warnings.filterwarnings("ignore")

except ImportError:
    colab = False

# Step 1: Load and Prepare Data
def load_data():
    if colab:
        print("Please upload the Excel file 'Table 21.1-1.xlsx' or 'Table 21.1-1.xls'")
        uploaded = files.upload()
        file_name = next(iter(uploaded))
        print(f"Uploaded file: {file_name}")
    else:
        file_name = 'Table 21.1-1.xlsx'  # Update with full path if needed
        print("Looking for 'Table 21.1-1.xlsx' in the current directory.")

    # Check file existence
    possible_extensions = ['.xlsx', '.xls', '.csv']
    file_path = file_name
    for ext in possible_extensions:
        test_path = file_name if file_name.endswith(ext) else file_name.replace('.xlsx', ext)
        if os.path.exists(test_path):
            file_path = test_path
            break
    else:
        print(f"Error: File '{file_name}' (or .xls/.csv variants) not found.")
        print("Current directory contents:", os.listdir('.'))
        raise FileNotFoundError(f"No file found matching '{file_name}'")

    # Load the data
    try:
        data = pd.read_excel(file_path, sheet_name='Table 21.1-1')
        print("\nFirst 5 rows of the loaded data:")
        print(data.head())
        print("\nAvailable columns in the data:", list(data.columns))
    except Exception as e:
        print(f"Error: Could not read the file. Current error: {e}")
        print("Check if the sheet 'Table 21.1-1' exists.")
        raise

    # Verify required columns
    required_columns = ['YEAR', 'DPI']
    if not all(col in data.columns for col in required_columns):
        print(f"Error: Required columns {required_columns} not found.")
        print("Available columns:", list(data.columns))
        raise ValueError("Missing required columns.")

    # Compute LOG_DPI if not present
    if 'LOG_DPI' not in data.columns:
        try:
            data['LOG_DPI'] = np.log(data['DPI'])
            print("\nComputed LOG_DPI as natural log of DPI.")
        except Exception as e:
            print(f"Error computing LOG_DPI from DPI: {e}")
            raise
    else:
        data['LOG_DPI'] = pd.to_numeric(data['LOG_DPI'], errors='coerce')
        if data['LOG_DPI'].isna().any():
            print("Warning: Some LOG_DPI values could not be converted to numeric.")
            raise ValueError("Invalid LOG_DPI values.")

    # Parse YEAR as quarterly dates (e.g., '1970.1'  '1970-01-01')
    try:
        data['YEAR'] = data['YEAR'].apply(lambda x: f"{int(float(x))}-{(int(float(x)*10)%10)*3:02d}-01")
        data['YEAR'] = pd.to_datetime(data['YEAR'], format='%Y-%m-%d')
        if data['YEAR'].isna().any():
            print("Error: Some dates in 'YEAR' could not be parsed.")
            raise ValueError("Invalid date format.")
        data.set_index('YEAR', inplace=True)
        series = data['LOG_DPI']
    except Exception as e:
        print(f"Error parsing 'YEAR' as dates: {e}")
        raise

    print("\nUsing 'YEAR' as index and 'LOG_DPI' as log DPI series.")
    print("First 5 rows of the final series:")
    print(series.head())
    return series

# Step 2: Check Stationarity with ADF Test
def adf_test(series, title=''):
    result = adfuller(series.dropna(), autolag='AIC')
    print(f'ADF Test for {title}:')
    print(f'ADF Statistic: {result[0]:.4f}')
    print(f'p-value: {result[1]:.4f}')
    print('Critical Values:', {k: v for k, v in result[4].items()})
    if result[1] < 0.05:
        print('Series is stationary\n')
    else:
        print('Series is non-stationary\n')

adf_test(series, 'Log DPI')

# Step 3: Differencing if Non-Stationary
diff_series = series.diff().dropna()
adf_test(diff_series, 'First Differenced Log DPI')

# Plot differenced series
plt.figure(figsize=(10, 6))
plt.plot(diff_series, label='First Differenced Log DPI')
plt.title('First Differenced Log DPI')
plt.xlabel('Date')
plt.ylabel(' Log DPI')
plt.legend()
plt.show()

# Step 4: ACF and PACF Plots for Model Identification

plt.title('ACF of Differenced Log DPI')
plt.subplot(122)
plot_pacf(diff_series, lags=20, ax=plt.gca())
plt.title('PACF of Differenced Log DPI')
plt.tight_layout()

# Step 5: Fit ARIMA Model
# Example: ARIMA(1,1,1); adjust p, d, q based on ACF/PACF
model = ARIMA(series, order=(1, 1, 1))

# Step 6: Diagnostic Checking

# ACF of residuals

# Ljung-Box test for residual autocorrelation
lb_test = acorr_ljungbox(residuals, lags=[10], return_df=True)

print(lb_test)

# Step 7: Forecasting
# Forecast 8 periods ahead (2 years for quarterly data)
forecast = results.forecast(steps=8)
